FUNCTIONS

function R = myrect(M)
	R=zeros(0,M);
   for i=1:M
		R(i)=1;
   end
end

function R=mybartlett(M)
	R=zeros(0,M);
    for i=1:(M/2)
		R(i)=2*i/M;
    for k=(M/2):M
        R(k)=2-(2*k/M);
    end
    end
end

function R = myhanning(M)
	R=zeros(0,M);
	for i=1:M
		R(i)=0.5*(1-(cos(2*pi*i/(M-1))));
	end
end

function R = myhamming(M)
	R=zeros(0,M);
	for i=1:M
		R(i)=0.54-0.46*(cos(2*pi*i/(M-1)));
	end
end

function R = myblackman(M)
	R=zeros(0,M);
	for i=1:M
		R(i)=0.42-0.5*(cos(2*pi*i/(M-1)))+0.08*(cos(4*pi*i/(M-1)));
	end
end
%for odd M
function R=mybartlett2(M)
	 R=zeros(0,M);
     for i=1:((M+1)/2)
		 R(i)=(2*i/M);
     for k=((M+1)/2):M
         R(k)=2-(2*k/M);
     end
     end
end
